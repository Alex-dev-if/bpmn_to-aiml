# Como  colocar para funcionar localmente
- mkdir chatbot
- cd chatbot
- git clone repositorio
- cd pasta_do_repositorio
- pip instal virtualenv
- virtualenv venv
- venv\script\activate.bat
- pip install -r requirements.txt
- python clientwebhack.py
